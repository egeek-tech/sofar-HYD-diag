```go
func (h *Hub) handleSelectPack(cmd ClientCommand) {
    msg := cmd.Message
    input := msg.Input
    tower := msg.Tower
    pack := msg.Pack

    // Validate and clamp to topology bounds (T-05-02)
    if input < 1 { input = 1 }
    if input > h.defaultBatInputs { input = h.defaultBatInputs }
    if tower < 1 { tower = 1 }
    if tower > h.defaultBatTowers { tower = h.defaultBatTowers }
    if pack < 1 { pack = 1 }
    if pack > h.defaultBatPacks { pack = h.defaultBatPacks }

    // Store selection for auto-refresh
    h.selectedPack = &packSelection{input: input, tower: tower, pack: pack, client: cmd.Client}

    h.triggerPackRead(input, tower, pack, cmd.Client)
}
```

Implement triggerPackRead:
```go
func (h *Hub) triggerPackRead(input, tower, pack int, client *Client) {
    towersPerInput := h.defaultBatTowers
    queryWord := register.EncodePackQuery(input, tower, pack, towersPerInput)

    // Build read requests from probe definitions
    rtProbes := register.PackRTProbes()
    infoProbes := register.PackInfoProbes()
    temps58Probes := register.PackTemps58Probes()

    rtReads := make([]broker.ReadRequest, 0)
    // Pack RT is a contiguous block 0x9044-0x907F = 60 registers, single batch read
    rtReads = append(rtReads, broker.ReadRequest{Addr: 0x9044, Count: 60})

    infoReads := make([]broker.ReadRequest, 0)
    // Pack Info block 0x9104-0x9126 = 35 registers
    infoReads = append(infoReads, broker.ReadRequest{Addr: 0x9104, Count: 35})

    // Temps 5-8 = 4 registers
    temps58Reads := make([]broker.ReadRequest, 0)
    temps58Reads = append(temps58Reads, broker.ReadRequest{Addr: 0x90BC, Count: 4})

    go func() {
        // Step 1: Write 0x9020 to select pack (D-01: function 0x10 via WriteRegister)
        err := h.broker.WriteRegister(h.ctx, 0x9020, queryWord)
        if err != nil {
            h.logger.Warn("pack select write failed, retrying with 2s settle", "error", err)
            // Retry once with 2s settle (D-01)
            time.Sleep(2 * time.Second)
            err = h.broker.WriteRegister(h.ctx, 0x9020, queryWord)
            if err != nil {
                h.logger.Error("pack select write failed after retry", "error", err)
                h.sendPackError(client, input, tower, pack, "timeout writing pack selection after retry")
                return
            }
        }

        // Step 2: Wait 1s settle time (D-01)
        time.Sleep(1 * time.Second)

        // Step 3: Read Pack RT data (0x9044-0x907F)
        rtResults := h.broker.ReadBatch(h.ctx, rtReads)
        if len(rtResults) == 0 || rtResults[0].Err != nil {
            errMsg := "timeout reading pack RT data after retry"
            if len(rtResults) > 0 && rtResults[0].Err != nil {
                errMsg = rtResults[0].Err.Error()
            }
            h.sendPackError(client, input, tower, pack, errMsg)
            return
        }

        // Step 4: Read Pack Info (0x9104-0x9126)
        infoResults := h.broker.ReadBatch(h.ctx, infoReads)

        // Step 5: Read Temps 5-8 (0x90BC-0x90BF)
        temps58Results := h.broker.ReadBatch(h.ctx, temps58Reads)

        // Step 6: Build and send pack data message
        msg := h.buildPackDataMessage(input, tower, pack, rtProbes, infoProbes, temps58Probes, rtResults, infoResults, temps58Results)
        h.sendToClient(client, msg)
    }()
}
```

Implement buildPackDataMessage - this is the core data assembly:
- Parse RT data (120 bytes for 60 registers) using probe definitions to extract values
- For the "Pack Info" group: format each probe using FormatValue into Items map
- For "Cell Voltages" group (type="cell_grid"): extract raw uint16 values for cells 1-24 from RT data at offsets for 0x9051-0x9068. Also extract max (0x9069) and min (0x906A) register values. Compute max/min cell index by scanning the 24 values. Pack into Cells []int (millivolt values), MaxCell, MinCell, MaxCellIndex, MinCellIndex (1-based).
- For "Temperatures" group: format each temp using FormatValue into Items. Also build TempRaw []int with raw S16 values (for frontend color coding: value is raw register value, e.g. 285 = 28.5C).
- For "Pack Status" group (type="pack_status"): read alarm/protection/fault register raw values from RT data and Pack Info. Decode bitmaps using DecodeBMSBitmap with the alarm/protection/fault tables. Populate Decoded []string with all decoded descriptions.
- For "Balance State" group (type="balance"): read 0x9075 raw value, put in BalanceBitmap.

Helper to extract uint16 at an address offset from the RT block:
```go
func extractU16(data []byte, baseAddr, targetAddr uint16) uint16 {
    offset := int(targetAddr-baseAddr) * 2
    if offset < 0 || offset+2 > len(data) {
        return 0
    }
    return binary.BigEndian.Uint16(data[offset : offset+2])
}
```

Helper to extract S16:
```go
func extractS16(data []byte, baseAddr, targetAddr uint16) int16 {
    return int16(extractU16(data, baseAddr, targetAddr))
}
```
