package hub

import "time"

// WebSocket message type constants (D-02, D-07, D-17, D-23, D-26)
const (
    MsgTypeSubscribe   = "subscribe"
    MsgTypeUnsubscribe = "unsubscribe"
    MsgTypeConnect     = "connect"
    MsgTypeDisconnect  = "disconnect"
    MsgTypeRefresh     = "refresh"
    MsgTypeAutoRefresh = "auto_refresh"
    MsgTypeSectionData = "section_data"
    MsgTypeSectionErr  = "section_error"
    MsgTypeState       = "connection_state"
)

// InboundMessage represents a message from client to server (D-07).
// Sent via WebSocket as JSON. Type determines which fields are relevant.
type InboundMessage struct {
    Type    string `json:"type"`
    Section string `json:"section,omitempty"`
    // Connect-specific fields (D-11)
    Host    string `json:"host,omitempty"`
    Port    int    `json:"port,omitempty"`
    SlaveID int    `json:"slave_id,omitempty"`
    // Auto-refresh toggle (D-35)
    Enabled *bool  `json:"enabled,omitempty"`
}

// OutboundMessage represents a message from server to client (D-02).
// Type + Section identify the message; Data carries pre-formatted key-value pairs.
type OutboundMessage struct {
    Type      string            `json:"type"`
    Section   string            `json:"section,omitempty"`
    Data      map[string]string `json:"data,omitempty"`
    State     string            `json:"state,omitempty"`
    Error     string            `json:"error,omitempty"`
    Timestamp string            `json:"timestamp,omitempty"`
}

// NewSectionData creates a section_data outbound message (D-26, D-10).
func NewSectionData(section string, data map[string]string) OutboundMessage {
    return OutboundMessage{
        Type:      MsgTypeSectionData,
        Section:   section,
        Data:      data,
        Timestamp: time.Now().UTC().Format(time.RFC3339),
    }
}

// NewSectionError creates a section_error outbound message (D-09).
func NewSectionError(section string, errMsg string) OutboundMessage {
    return OutboundMessage{
        Type:      MsgTypeSectionErr,
        Section:   section,
        Error:     errMsg,
        Timestamp: time.Now().UTC().Format(time.RFC3339),
    }
}

// NewStateMessage creates a connection_state outbound message (D-04).
func NewStateMessage(state string) OutboundMessage {
    return OutboundMessage{
        Type:  MsgTypeState,
        State: state,
    }
}
```

    // StateEvents returns a channel emitting connection state changes (D-04).
    StateEvents() <-chan broker.StateEvent
}
```
