<objective>
Implement immediate disconnect response that aborts in-progress Modbus reads within 1 second.

Purpose: When a user clicks Disconnect while a read cycle is in progress, the blocking TCP socket read must be interrupted immediately. Currently, the broker's command channel is single-consumer -- if executeRead is blocking on a 10-second TCP read, the CmdDisconnect command cannot be dequeued until that read completes. This plan adds an abort mechanism that shortcuts the blocking read by setting the read deadline to now.

Output: Modified broker with abortRead()/connMu, modified hub disconnect handler that cancels section reads first, and tests verifying <1s disconnect during active reads.
</objective>

3. Add the `abortRead()` method (per D-02). Place it after the `Close()` method, before `Reconfigure()`:
```go
// abortRead forces any in-progress socket read to fail immediately
// by setting the read deadline to now. Safe to call from any goroutine.
// Must be non-blocking to avoid deadlock with the Run() loop.
func (b *Broker) abortRead() {
    b.connMu.Lock()
    defer b.connMu.Unlock()
    if b.conn != nil {
        b.conn.SetReadDeadline(time.Now())
    }
}
```

4. Modify the `Disconnect()` method to call `abortRead()` BEFORE queuing the command on the channel (per D-02, RESEARCH Pattern 1). This ensures the blocking TCP read is interrupted so the command channel can drain:
```go
func (b *Broker) Disconnect(ctx context.Context) error {
    b.abortRead() // unblock any pending read FIRST (D-02)

    respCh := make(chan interface{}, 1)
    select {
    case b.commands <- command{
        cmdType:  CmdDisconnect,
        request:  nil,
        response: respCh,

    case <-b.done:
        return ErrBrokerClosed
    case <-ctx.Done():
        return ctx.Err()
    }
    select {
    case <-respCh:
        return nil
    case <-ctx.Done():
        return ctx.Err()
    }
}
```

8. Protect conn assignment in `executeReconfigure()` with connMu:
```go
func (b *Broker) executeReconfigure(_ context.Context, req ReconfigureRequest) {
    // Close existing connection if any
    b.connMu.Lock()
    if b.conn != nil {
        b.conn.Close()
        b.conn = nil
    }
    b.connMu.Unlock()
    b.addr = req.Addr
    b.slaveID = req.SlaveID
    b.dormant = false
    b.backoff.Reset()
    b.logger.Info("broker reconfigured", "addr", req.Addr, "slaveID", req.SlaveID)

    b.setState(StateConnecting, nil)
    conn, err := modbus.Connect(b.addr)
    if err != nil {
        b.setState(StateDisconnected, err)
        b.logger.Error("connection failed", "addr", b.addr, "error", err)
        return
    }
    b.connMu.Lock()
    b.conn = conn
    b.connMu.Unlock()
    b.setState(StateConnected, nil)
}
```

NOTE: Inside `executeRead`, `executeWrite`, `executeBatch`, and `ensureConnected`, the conn field is accessed only from the Run() goroutine. Since `abortRead()` only sets the deadline (does not close/replace), and `executeDisconnect`/`handleError`/`cleanup`/`executeReconfigure` now hold connMu when closing/replacing conn, the mutex protects against the only real race: abortRead acting on a conn that is simultaneously being closed. Inside the Run loop, conn reads for data do NOT need the mutex because the Run goroutine is the only reader, and SetReadDeadline from abortRead is safe to call concurrently with Read per Go net.Conn docs.

```go
    for _, sec := range h.sections {
        sec.cancelRead()
    }

        if err := h.broker.Disconnect(h.ctx); err != nil {
            h.logger.Error("disconnect failed", "error", err)
        }
    }()
```

This ensures that:
1. All section readCtx values are cancelled immediately (streaming goroutines check readCtx.Err() between probes)
2. broker.Disconnect() then calls abortRead() to unblock any pending TCP read
3. The CmdDisconnect command processes quickly since the read loop has already exited
4. The broker emits StateDisconnected which the hub broadcasts to clients

The order is: cancel contexts (hub) -> abortRead (broker, from Disconnect) -> CmdDisconnect queued -> executeDisconnect runs -> StateDisconnected emitted -> client notified.

Note: The handleStateEvent already cancels all reads on StateDisconnected and StateReconnecting (line 249-253), so the explicit cancel here is an optimization for speed -- it ensures reads are cancelled even before the state event arrives. The double cancellation is idempotent since cancelRead() is a no-op if readCancel is nil.

```go
func TestBrokerAbortRead(t *testing.T) {
    // Start a slow mock server that holds the connection for 10s
    listener, err := net.Listen("tcp", "127.0.0.1:0")
    if err != nil {
        t.Fatalf("failed to create listener: %v", err)
    }
    defer listener.Close()
    addr := listener.Addr().String()
    go slowMockServer(t, listener, 10*time.Second)

    ctx, cancel := context.WithTimeout(context.Background(), 15*time.Second)
    defer cancel()

    b := broker.New(discardLogger(), "", 1, false)
    b.SetInterReadDelay(10 * time.Millisecond)
    go b.Run(ctx)
    defer b.Close()

    // Connect to the slow server
    if err := b.Reconfigure(ctx, addr, 1); err != nil {
        t.Fatalf("Reconfigure failed: %v", err)
    }

    // Start a read in background -- this will block on the slow server
    readDone := make(chan error, 1)
    go func() {
        _, err := b.ReadRegisters(ctx, 0x0404, 1)
        readDone <- err
    }()

    // Give the read a moment to start blocking on TCP
    time.Sleep(200 * time.Millisecond)

    // Disconnect should abort the blocking read and complete quickly
    start := time.Now()
    if err := b.Disconnect(ctx); err != nil {
        t.Fatalf("Disconnect failed: %v", err)
    }
    elapsed := time.Since(start)

    // Disconnect must complete within 2 seconds (D-02 requires <1s, allow margin)
    if elapsed > 2*time.Second {
        t.Errorf("Disconnect took %v, expected < 2s", elapsed)
    }

    // Broker should be disconnected
    if s := b.CurrentState(); s != broker.StateDisconnected {
        t.Errorf("expected StateDisconnected, got %v", s)
    }

    // The blocked read should also have returned (with an error)
    select {
    case err := <-readDone:
        if err == nil {
            t.Error("expected read to fail after abort, got nil error")
        }
    case <-time.After(3 * time.Second):
        t.Error("read goroutine did not return within 3s after disconnect")
    }
}
```
